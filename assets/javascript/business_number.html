# 数字处理

numbro.js

  1. 千分位处理
const thousandSeparated = (value) => {
  // 将数字转为string类型，并且拆分整数部分、小数部分和负数符号
  let result = value.toString();
  let characteristic = result.split(".")[0];
  let mantissa = result.split(".")[1];
  const hasNegativeSign = value < 0 && characteristic.indexOf("-") === 0;

  if (hasNegativeSign) {
    // 删除负数符号
    characteristic = characteristic.slice(1);
  }
  //  计算出插入分隔符号的位置。 🎈:字符串从左至右方向的index
  let indexesToInsertThousandDelimiters = indexesOfGroupSpace(
    characteristic.length,
    3
  );
  // 插入分隔符。🎈:position + index，因为index和插入后的字符串的position增长一致
  indexesToInsertThousandDelimiters.forEach((position, index) => {
    characteristic =
      characteristic.slice(0, position + index) +
      "," +
      characteristic.slice(position + index);
  });
  // 拼装回符号和小数部分
  if (hasNegativeSign) {
    characteristic = `-${characteristic}`;
  }
  if (!mantissa) {
    result = characteristic;
  } else {
    result = characteristic + "." + mantissa;
  }
  return result;
};

function indexesOfGroupSpace(totalLength, groupSize) {
  let result = [];
  let counter = 0;
  for (let i = totalLength; i > 0; i--) {
    if (counter === groupSize) {
      result.unshift(i);
      counter = 0;
    }
    counter++;
  }
  return result;
}
