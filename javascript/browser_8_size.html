# 窗口、显示器、文档大小

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>窗体大小</title>
    <style>
        html,
        body {
            margin: 0;
            padding: 0;
            position: relative;
        }

        #wrap {
            position: relative;
            padding: 20px;
            border: 1px solid blue;
            background-color: cadetblue;
        }

        #box {
            width: 900px;
            height: 1300px;
            margin: 20px;
            border: 10px solid lightblue;
            background-color: cornflowerblue;
            padding: 20px;
        }

        dl {
            color: #ffffff;
        }

        dt {
            line-height: 2em;
        }

        #tips {
            position: fixed;
            width: 500px;
            top: 80px;
            right: 0;
        }
    </style>
</head>
<body>
    <div id="wrap">
        <div class="content"></div>
        <div id="box">
        </div>
    </div>
    <div id="tips"></div>
    <script>
        function appendChild(wrap) {
            return function(innerText, tag = "dd") {
                let node = document.createElement(tag)
                node.innerText = innerText;
                wrap.appendChild(node)
            }
        }
        let $dl = document.createElement("dl")
        const append = appendChild($dl)
        append("浏览器 Window", "dt")
        append(`浏览器窗体的大小，包括任务栏 window.outerWidth/outerHeight: (${window.outerWidth},${window.outerHeight})`)
        append(`浏览器内部大小，不包括用户界面元素 window.innerWidth/innerHeight: (${window.innerWidth},${window.innerHeight})`)
        append(`网页滚动的像素值，x方向和y方向 window.pageXOffset/pageYOffset: (${window.pageXOffset},${window.pageYOffset})`)
        append(`浏览器窗口在显示器中的位置 window.screenX(screenLeft)/screenY(screenTop): (${window.screenX},${window.screenY})`)
        append("显示器 Screen", "dt")
        append(` screen.width/height (${screen.width}, ${screen.height})`)
        append(`可用宽高 screen.availWidth/availHeight (${screen.availWidth}, ${screen.availHeight})`)
        append(`颜色深度 screen.colorDepth (${screen.colorDepth})`)
        append(` screen.pixelDepth (${screen.pixelDepth})`)
        append("文档 document", "dt")
        append(`返回坐标处所在的元素。document.elementFromPoint(x,y)`)
        append(`id为box的元素距离文档左上角的矩形范围。注意，滚动的时候，会有负值现象。 box.getBoundingClientRect ${box.getBoundingClientRect()}`)
        append(`返回inline-box 的矩形区域。是个数组结构。 box.getClientRects `)
        // getClientRects和getBoundingClientRect差异
        // getClientRects返回的其实是个数组，数组中有很多个类似getBoundingClientRect返回的对象。
        // getBoundingClientRect返回的永远是最外框框的那个矩形区域相关的坐标偏移对象；
        // 而getClientRects是多行文字区域的坐标偏移集合，在非IE浏览器下，只对inline水平的标签有反应。
        // 一般getBoundingClientRect方法用的多一点。我们可以很容易获取某个元素的偏移值。甚至高宽都能很轻松的计算出来。
        // 所以，下载你想用js获取元素的高宽尺寸，就可以试试getBoundingClientRect方法了。
        document.getElementById("tips").appendChild($dl)
    </script>
</body>
</html>
