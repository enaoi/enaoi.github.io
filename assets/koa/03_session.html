# Session 操作

koa2没有提供session 操作，只能通过第三方中间件。 以 koa-session 为例：

config 的配置参数大多和cookie类似，另外说明：

  • rolling： 在每次请求时强行设置cookie，这将重置cookie过期时间，默认值为false
  • renew： 当seesion 快要过期的时候，我们可以一直保持用户登录状态

默认 session 存放在cookie

const Koa = require('koa')
const session = require('koa-session')
const app = new Koa()

// asigned 为true时，必须要设置app.keys
app.keys = ['awesomekeys']

const CONFIG = {
  key: 'koa.sess', // 默认值
  maxAge: 10000, // 默认是1day
  autoCommit: true,
  overwrite: true,
  httpOnly: true,
  signed: true,
  rolling: false,
  renew: false,
  secure: false,
  sameSite: null,
  beforeSave(ctx, session) {
    return session;
  },
  valid(ctx, session) {
    return session
  },
  encode: json => JSON.stringify(json),
  decode: str => JSON.parse(str)
}

app.use(session(CONFIG, app))

app.use(ctx => {
  let n = ctx.session.views || 0;
  if (ctx.path === '/session') {
    ctx.session.views = ++n;
    ctx.body = n + ' views'
  } else {
    ctx.body = n + ' views'
  }
})

app.listen(3000, () => {
  console.log('server is running at port 3000')

永久存储：Redis、MongoDB、等。这里已文件形式存储

const store = {
  get(key) {
    const sessionDir = path.resolve(__dirname, './session')
    const files = fs.readdirSync(sessionDir)
    for (let i = 0; i < files.length; i++) {
      if (files[i].startsWith(key)) {
        const filePath = path.resolve(sessionDir, files[i])
        delete require.cache[require.resolve(filePath)]
        const result = require(filePath)
        return result;
      }
    }
  },
  async set(key, session) {
    const filePath = path.resolve(__dirname, './session', `${key}.js`)
    const content = `module.exports = ${JSON.stringify(session)}`
    fs.writeFileSync(filePath, content)
  },
  async destroy(key) {
    const filePath = path.resolve(__dirname, './session', `${key}.js`)
    fs.unlinkSync(filePath)
  }
}

const CONFIG= {
   ...,
      store
}
